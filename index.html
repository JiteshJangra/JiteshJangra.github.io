async resolveUrl(code) {
  const cached = await
    redis.get(`url:${'{code}'}`);
 
  if (cached) {
    // <5ms cache hit
    return { url: cached,
      fromCache: true };
  }
 
  // MongoDB fallback
  const url = await
    Url.findOne({ code });
  redis.setex(code, 86400, url);
  return { url, fromCache: false };
}
